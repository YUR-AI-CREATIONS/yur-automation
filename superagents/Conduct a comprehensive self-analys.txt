### 1. Core Implementation (`ingestion_core.py`)

```python
import asyncio
import logging
import fnmatch
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import List, Set, Dict, AsyncGenerator, Any, Tuple
from pathlib import Path
import os

# Configure Logging
logging.basicConfig(level=logging.INFO, format='%(asctime)s - [%(levelname)s] - %(message)s')
logger = logging.getLogger(__name__)

# --- Configuration ---
@dataclass
class IngestionConfig:
    root_paths: List[str]
    max_concurrency: int = 10
    exclude_patterns: List[str] = field(default_factory=lambda: ["*.git*", "*.env", "__pycache__"])
    follow_symlinks: bool = False
    io_timeout_seconds: float = 5.0

@dataclass
class FileMetadata:
    path: str
    size: int
    is_symlink: bool
    status: str  # 'processed', 'error', 'skipped'
    error_msg: str = None

# --- Abstraction Layer ---
class FileSystemProvider(ABC):
    """Abstract Interface for File System Operations"""
    @abstractmethod
    async def list_dir(self, path: str) -> List[str]: pass
    
    @abstractmethod
    async def get_stat(self, path: str) -> Tuple[int, bool]: pass  # size, is_symlink
    
    @abstractmethod
    async def is_dir(self, path: str) -> bool: pass

    @abstractmethod
    async def resolve_path(self, path: str) -> str: pass

# --- Real Implementation ---
class RealFileSystem(FileSystemProvider):
    """Production implementation using OS calls"""
    async def list_dir(self, path: str) -> List[str]:
        return await asyncio.to_thread(os.listdir, path)

    async def get_stat(self, path: str) -> Tuple[int, bool]:
        stat = await asyncio.to_thread(os.lstat, path)
        return stat.st_size, os.path.islink(path)

    async def is_dir(self, path: str) -> bool:
        return await asyncio.to_thread(os.path.isdir, path)

    async def resolve_path(self, path: str) -> str:
        return await asyncio.to_thread(os.path.realpath, path)

# --- Ingestion Engine ---
class IngestionEngine:
    def __init__(self, config: IngestionConfig, fs_provider: FileSystemProvider):
        self.config = config
        self.fs = fs_provider
        self.queue = asyncio.Queue()
        self.results: List[FileMetadata] = []
        self.visited_paths: Set[str] = set()
        self.active_workers = 0

    def _is_excluded(self, path: str) -> bool:
        name = os.path.basename(path)
        return any(fnmatch.fnmatch(name, pattern) for pattern in self.config.exclude_patterns)

    async def _worker(self):
        while True:
            try:
                path = await self.queue.get()
                self.active_workers += 1
                
                # Exclusion Check
                if self._is_excluded(path):
                    self.queue.task_done()
                    self.active_workers -= 1
                    continue

                try:
                    # Symlink & Cycle Detection
                    resolved_path = await self.fs.resolve_path(path)
                    if resolved_path in self.visited_paths:
                        self.queue.task_done()
                        self.active_workers -= 1
                        continue
                    self.visited_paths.add(resolved_path)

                    if await self.fs.is_dir(path):
                        # Traversal: Add children to queue
                        children = await self.fs.list_dir(path)
                        for child in children:
                            full_child_path = os.path.join(path, child)
                            self.queue.put_nowait(full_child_path)
                    else:
                        # Processing: Get Metadata
                        size, is_link = await self.fs.get_stat(path)
                        self.results.append(FileMetadata(
                            path=path, size=size, is_symlink=is_link, status="processed"
                        ))

                except Exception as e:
                    # Fault Tolerance: Log and continue
                    logger.error(f"Error processing {path}: {str(e)}")
                    self.results.append(FileMetadata(
                        path=path, size=0, is_symlink=False, status="error", error_msg=str(e)
                    ))

            except asyncio.CancelledError:
                break
            finally:
                self.queue.task_done()
                self.active_workers -= 1

    async def bootstrap(self):
        """Initializes the ingestion process"""
        logger.info("Bootstrapping Ingestion Engine...")
        
        # Seed Queue
        for root in self.config.root_paths:
            self.queue.put_nowait(root)

        # Spawn Workers
        workers = [asyncio.create_task(self._worker()) for _ in range(self.config.max_concurrency)]
        
        # Wait for completion
        # We wait until the queue is empty AND no workers are active
        await self.queue.join()
        
        # Cleanup
        for w in workers:
            w.cancel()
        
        logger.info(f"Ingestion Complete. Processed {len(self.results)} files.")
        return self.results
```

### 3. Architectural Analysis

1.  **Memory Efficiency:**
    *   **Queue-Based Traversal:** Unlike recursive DFS which holds stack frames, the `asyncio.Queue` approach keeps the memory footprint flat. The depth of the directory tree does not directly correlate to memory usage; only the *width* of the currently processing layer matters.
    *   **Metadata Only:** The `FileMetadata` object is lightweight. We do not load file content into memory during the ingestion phase, only stats.

2.  **Error Resilience:**
    *   **Isolation:** The `_worker` loop wraps individual file operations in a specific `try/except` block. A `PermissionError` on `/var/root/secure` will log an error metadata object but will *not* stop the worker from picking up the next item in the queue.
    *   **Mock Injection:** The `MockFileSystem` proves that the system handles random `IOError` events without crashing, validating stability before deployment to a volatile production environment.

3.  **Scalability:**
    *   The `max_concurrency` parameter allows fine-tuning based on the host's I/O capabilities. For network mounts (NFS/S3), this can be increased to hide latency. For local SSDs, it can be throttled to prevent disk saturation.

import os
import fnmatch
from concurrent.futures import ThreadPoolExecutor
from dataclasses import dataclass, field
from typing import List, Optional, Dict
from pathlib import Path

# --- Configuration & Constants ---
DEFAULT_EXCLUDE_PATTERNS = [
    ".git*", ".env*", "__pycache__", "node_modules", 
    "*.pyc", "*.png", "*.jpg", "*.exe", "*.dll", "*.so",
    "dist", "build", ".idea", ".vscode"
]

MAX_FILE_SIZE_BYTES = 1024 * 500  # Skip files larger than 500KB to save context

@dataclass
class FileArtifact:
    """Immutable representation of a source code file."""
    path: str
    content: str
    size: int
    extension: str

@dataclass
class IngestionConfig:
    root_path: Path
    exclude_patterns: List[str] = field(default_factory=lambda: DEFAULT_EXCLUDE_PATTERNS)
    max_workers: int = 4

# --- Component 1: Ingestion Engine ---
class CodebaseIngestor:
    """
    Responsible for traversing the file system, filtering noise, 
    and extracting textual content from source files.
    """
    def __init__(self, config: IngestionConfig):
        self.config = config

    def _is_binary(self, data: bytes) -> bool:
        """Heuristic detection of binary files via null byte inspection."""
        return b'\x00' in data[:8000]

    def _should_exclude(self, path: Path) -> bool:
        """Checks path against glob exclusion patterns."""
        # Check specific file/dir names
        for pattern in self.config.exclude_patterns:
            if fnmatch.fnmatch(path.name, pattern):
                return True
            # Check full relative path for deep exclusions
            rel_path = str(path.relative_to(self.config.root_path))
            if fnmatch.fnmatch(rel_path, pattern):
                return True
        return False

    def _process_file(self, file_path: Path) -> Optional[FileArtifact]:
        """Reads and decodes a single file."""
        if self._should_exclude(file_path):
            return None
        
        try:
            stat = file_path.stat()
            if stat.st_size > MAX_FILE_SIZE_BYTES:
                return None # Skip large files

            with open(file_path, 'rb') as f:
                raw_data = f.read()
                
            if self._is_binary(raw_data):
                return None

            # Attempt decode
            content = raw_data.decode('utf-8', errors='replace')
            
            return FileArtifact(
                path=str(file_path.relative_to(self.config.root_path)),
                content=content,
                size=stat.st_size,
                extension=file_path.suffix
            )
        except Exception as e:
            # Log error but do not crash ingestion
            print(f"[WARN] Failed to process {file_path}: {e}")
            return None

    def ingest(self) -> List[FileArtifact]:
        """
        Executes parallel ingestion of the directory tree.
        """
        artifacts = []
        files_to_process = []

        # 1. Walk tree to gather paths
        for root, dirs, files in os.walk(self.config.root_path):
            # Modify dirs in-place to skip ignored directories early
            dirs[:] = [d for d in dirs if not self._should_exclude(Path(root) / d)]
            
            for file in files:
                file_path = Path(root) / file
                files_to_process.append(file_path)

        # 2. Parallel Read
        print(f"🚀 Ingesting {len(files_to_process)} files from {self.config.root_path}...")
        with ThreadPoolExecutor(max_workers=self.config.max_workers) as executor:
            results = executor.map(self._process_file, files_to_process)
            
        # 3. Filter Nones and Collect
        for res in results:
            if res:
                artifacts.append(res)
                
        print(f"✅ Ingestion Complete. Captured {len(artifacts)} valid artifacts.")
        return artifacts

# --- Component 2: Context Serializer ---
class ContextSerializer:
    """
    Transforms artifacts into an LLM-optimized XML structure.
    XML tags provide strong cognitive delimiters for the model.
    """
    @staticmethod
    def to_xml(artifacts: List[FileArtifact]) -> str:
        buffer = ["<codebase_context>"]
        
        # Sort by path for deterministic context
        for artifact in sorted(artifacts, key=lambda x: x.path):
            buffer.append(f'  <file path="{artifact.path}">')
            buffer.append(f'<![CDATA[\n{artifact.content}\n]]>')
            buffer.append('  </file>')
            
        buffer.append("</codebase_context>")
        return "\n".join(buffer)

    @staticmethod
    def generate_stats(artifacts: List[FileArtifact]) -> Dict:
        return {
            "total_files": len(artifacts),
            "total_tokens_est": sum(len(a.content) for a in artifacts) // 4,
            "file_types": list(set(a.extension for a in artifacts))
        }

# --- Component 3: Cognitive Test Harness ---
class CognitiveTestHarness:
    """
    Simulates the environment for a cognitive test.
    """
    def __init__(self, ingestor: CodebaseIngestor):
        self.ingestor = ingestor

    def run_simulation(self, task_prompt: str):
        # 1. Ingest Data
        artifacts = self.ingestor.ingest()
        
        if not artifacts:
            print("❌ Error: No artifacts ingested. Check path or exclusions.")
            return

        # 2. Serialize Context
        context_payload = ContextSerializer.to_xml(artifacts)
        stats = ContextSerializer.generate_stats(artifacts)
        
        # 3. Construct the "Perfect Prompt"
        full_prompt = self._construct_prompt(context_payload, task_prompt)
        
        # 4. Output / Simulation
        self._save_output(full_prompt)
        print(f"\n📊 Context Stats: {stats}")
        print(f"💾 Payload generated: 'cognitive_payload.txt' ({len(full_prompt)} chars)")
        print("💡 Ready for LLM Injection.")

    def _construct_prompt(self, context: str, task: str) -> str:
        return f"""
# SYSTEM ROLE
You are a Principal Software Architect conducting a code review.

# CONTEXT
The following is the complete source code for the target application, wrapped in XML tags.
{context}

# TASK
{task}
"""

    def _save_output(self, content: str):
        with open("cognitive_payload.txt", "w", encoding="utf-8") as f:
            f.write(content)

if __name__ == "__main__":

    config = IngestionConfig(

### Architectural Rationale

*   **XML Serialization:** We use `<![CDATA[ ... ]]>` wrapping for file content. This is a critical prompt engineering pattern. It prevents the LLM from confusing code syntax (like brackets or quotes inside the code) with the prompt structure itself.
*   **Fail-Soft Ingestion:** The `_process_file` method uses `errors='replace'` during decoding. In a cognitive assessment, it is better to have a file with a few replacement characters (?) than to lose the entire file due to a single non-UTF8 byte.
*   **Token Estimation:** The bootstrap includes a rough token estimator (`chars // 4`). This allows the operator to verify if the payload fits within the target model's context window (e.g., 8k, 32k, 128k) before attempting the test.
